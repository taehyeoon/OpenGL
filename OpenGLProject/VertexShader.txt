#version 400 core

layout(location = 0) in vec3 inpos;
layout(location = 1) in vec3 inColor;
layout(location = 2) in vec2 inTexCoord;

out vec3 ourColor;
out vec2 texCoord;

uniform mat4 u_MVPMat;

void main()
{	
	gl_Position = u_MVPMat * vec4(inpos, 1);
    ourColor = inColor;
    texCoord = inTexCoord;
}
